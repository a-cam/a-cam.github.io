OpenGL

I've made some 2D games in the past using graphics engines such as SDL and monogame but I wanted to try and make my own wrapper around OpenGL if I ever wanted to make something in 3D.

The current version of my wrapper can load different mesh file types such as .obj with .mtl and display them with a simple lighting system as well.

The following screenshots are from a scene I created in blender from free 3D assets which I then exported as .obj and loaded into my program.

This is the scene lit from above.

And this is the scene with lighting from the side, the source represented as the white cube.

There are currently no shadows which is a feature I would like to add in the future.

The code to generate the scene would look something like this: 


						int main()
						{
							Renderer renderer;
						
							R_obj light_cube;
							light_cube.load_model("C:\\Path\\to\\unit_cube.obj", false);
							light_cube.set_world_location(glm::vec3(1.f, 1.f, 1.f));
							light_cube.set_is_affected_by_lighting(false);
							light_cube.build();
						
							Point_light point_light;
							point_light.set_world_location(glm::vec3(0.f, 1.f, 1.f));
							point_light.set_ambient_color(glm::vec3(0.2f, 0.2f, 0.2f));
							point_light.set_diffuse_color(glm::vec3(0.5f, 0.5f, 0.5f));
							point_light.set_specular_color(glm::vec3(0.f, 0.f, 0.f));
							point_light.set_attenuation_constants(Attenuation_constants(1.0, 0.0014, 0.000007));
						
							Directional_light sun_light;
							sun_light.set_ambient_color(glm::vec3(0.2f, 0.2f, 0.2f));
							sun_light.set_diffuse_color(glm::vec3(0.5f, 0.5f, 0.5f));
							sun_light.set_specular_color(glm::vec3(0, 0, 0));
							sun_light.set_direction(glm::vec3(0, 0, -1.f));
						
						
							R_obj scene;
							scene.load_model("C:\\Path\\to\\scene.obj", false);
							scene.set_is_affected_by_lighting(true);
							scene.set_world_location(glm::vec3(0.f, 0.f, 0.f));
							scene.build();
						
							int loop_counter = 0;
							while (!renderer.get_should_close_window())
							{
								loop_counter++;
						
								renderer.add_render_object(&light_cube);
						
								renderer.add_render_object(&scene);
								
								glm::vec3 light_pos = glm::vec3(
									0.f, 
									1.f + 50.f * cos((float)loop_counter/100.f), 
									1.f + 50.f * sin((float)loop_counter/100.f));
									
								point_light.set_world_location(light_pos);
								light_cube.set_world_location(light_pos);
						
								renderer.add_light(&sun_light);
								renderer.add_light(&point_light);
								
								renderer.update();
							}
						
							renderer.clean_up();
							return 0;
						}
